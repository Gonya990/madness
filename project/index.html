Тёплая дизайн-система для разработчиков — токены, 25 компонентов и продуктовые киты, чтобы писать, запускать и деплоить код вместе.

// Стримим токены обратно клиенту
import { serve } from '@unicode/sdk';

serve(async (req) => {
  const { prompt } = await req.json();
  return run('gpt', prompt, { stream: true });
});
